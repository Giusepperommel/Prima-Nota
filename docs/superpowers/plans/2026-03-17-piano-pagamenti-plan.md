# Piano Pagamenti Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Introduce installment payment tracking (PianoPagamento + Pagamento) to separate cash flow from fiscal competence, fixing the inflated cash outflow when assets are purchased in installments.

**Architecture:** Two new Prisma models (PianoPagamento, Pagamento) linked 1:1 to Operazione. Operations without a PianoPagamento are treated as immediate payments (backward compatible). The cash simulation dashboard reads from Pagamento records instead of Operazione.importoTotale. A new `calcolaPianoPagamento` utility generates installment schedules reusing the existing `calcolaPianoFinanziamento` logic from `calcoli-veicoli.ts`.

**Tech Stack:** Next.js 16, Prisma (MySQL), React 19, TypeScript, shadcn/ui, recharts, Zod, date-fns

**Spec:** `docs/superpowers/specs/2026-03-17-piano-pagamenti-design.md`

This file reuses the French amortization logic from `calcoli-veicoli.ts` and adds schedule generation with dates.

```typescript
/**
 * Payment plan calculation utilities.
 * Generates installment schedules (piano rate) with dates,
 * capital/interest split, and validation for custom payments.
 */

import { addMonths } from "date-fns";
import { calcolaPianoFinanziamento } from "./calcoli-veicoli";

// ---------------------------------------------------------------------------
// Types
// ---------------------------------------------------------------------------

export type RataPagamento = {
  numeroPagamento: number;
  data: Date;
  importo: number;
  quotaCapitale: number;
  quotaInteressi: number;
};

export type PianoPagamentoGenerato = {
  rate: RataPagamento[];
  totaleInteressi: number;
  totaleCapitale: number;
  importoRata: number;
};

// ---------------------------------------------------------------------------
// Schedule generation
// ---------------------------------------------------------------------------

/**
 * Generates a full installment payment schedule with dates.
 * Reuses calcolaPianoFinanziamento from calcoli-veicoli.ts for the
 * French amortization math, then adds date assignment.
 *
 * @param importoDaFinanziare - Amount to finance (total - anticipo)
 * @param numeroRate - Number of installments
 * @param tan - Annual nominal rate (percentage, e.g. 5.5 for 5.5%). 0 or null for interest-free.
 * @param dataInizio - Date of the first installment
 * @returns Generated schedule with dates and capital/interest split
 */
export function generaPianoPagamento(
  importoDaFinanziare: number,
  numeroRate: number,
  tan: number | null | undefined,
  dataInizio: Date
): PianoPagamentoGenerato {
  if (numeroRate <= 0 || importoDaFinanziare <= 0) {
    return { rate: [], totaleInteressi: 0, totaleCapitale: 0, importoRata: 0 };
  }

  // Calculate importoRata for calcolaPianoFinanziamento
  const haInteressi = tan != null && tan > 0;
  let importoRata: number;
  if (haInteressi) {
    const tassoMensile = tan / 100 / 12;
    const factor = Math.pow(1 + tassoMensile, numeroRate);
    importoRata = Math.round(
      (importoDaFinanziare * (tassoMensile * factor) / (factor - 1)) * 100
    ) / 100;
  } else {
    importoRata = Math.round((importoDaFinanziare / numeroRate) * 100) / 100;
  }

  // Reuse existing amortization logic
  const pianoFin = calcolaPianoFinanziamento(importoDaFinanziare, numeroRate, importoRata, tan);

  // Add dates
  const rate: RataPagamento[] = pianoFin.map((r) => ({
    numeroPagamento: r.numeroRata,
    data: addMonths(dataInizio, r.numeroRata - 1),
    importo: r.importoRata,
    quotaCapitale: r.quotaCapitale,
    quotaInteressi: r.quotaInteressi,
  }));

  const totaleInteressi = Math.round(
    rate.reduce((sum, r) => sum + r.quotaInteressi, 0) * 100
  ) / 100;

  return {
    rate,
    totaleInteressi,
    totaleCapitale: importoDaFinanziare,
    importoRata,
  };
}

// ---------------------------------------------------------------------------
// Validation
// ---------------------------------------------------------------------------

/**
 * Validates that custom payment amounts don't exceed the operation total.
 * @returns The remaining uncovered amount (0 if fully covered, negative if over).
 */
export function validaPagamentiCustom(
  importoTotaleOperazione: number,
  pagamenti: { importo: number }[]
): { copertura: number; rimanente: number; superato: boolean } {
  const copertura = pagamenti.reduce((sum, p) => sum + p.importo, 0);
  const rimanente = Math.round((importoTotaleOperazione - copertura) * 100) / 100;
  return {
    copertura: Math.round(copertura * 100) / 100,
    rimanente,
    superato: rimanente < 0,
  };
}
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { auth } from "@/lib/auth";
import { prisma } from "@/lib/prisma";
import { generaPianoPagamento } from "@/lib/calcoli-pagamenti";

export async function POST(request: NextRequest) {
  try {
    const session = await auth();
    if (!session?.user) {
      return NextResponse.json({ error: "Non autenticato" }, { status: 401 });
    }

    const user = session.user as any;
    const societaId = user.societaId as number;

    const body = await request.json();
    const {
      operazioneId,
      tipo, // "RATEALE" | "CUSTOM"
      numeroRate,
      tan,
      anticipo,
      dataInizio,
      // For CUSTOM mode:
      pagamentiCustom, // Array<{ data: string, importo: number, note?: string }>
    } = body;

    if (!operazioneId || !tipo || !dataInizio) {
      return NextResponse.json(
        { error: "Campi obbligatori: operazioneId, tipo, dataInizio" },
        { status: 400 }
      );
    }

    // Verify operazione exists and belongs to this society
    const operazione = await prisma.operazione.findFirst({
      where: { id: operazioneId, societaId, eliminato: false },
      select: { id: true, importoTotale: true, pianoPagamento: { select: { id: true } } },
    });

    if (!operazione) {
      return NextResponse.json({ error: "Operazione non trovata" }, { status: 404 });
    }

    if (operazione.pianoPagamento) {
      return NextResponse.json(
        { error: "L'operazione ha già un piano di pagamento" },
        { status: 409 }
      );
    }

    const importoTotale = Number(operazione.importoTotale);

    const result = await prisma.$transaction(async (tx) => {
      if (tipo === "RATEALE") {
        if (!numeroRate || numeroRate <= 0) {
          throw new Error("Numero rate obbligatorio per piano rateale");
        }

        const anticipoVal = anticipo ? parseFloat(String(anticipo)) : 0;
        const importoDaFinanziare = importoTotale - anticipoVal;

        if (importoDaFinanziare <= 0) {
          throw new Error("L'anticipo non può essere maggiore o uguale all'importo totale");
        }

        const tanVal = tan != null ? parseFloat(String(tan)) : 0;

        const piano = generaPianoPagamento(
          importoDaFinanziare,
          parseInt(String(numeroRate)),
          tanVal,
          new Date(dataInizio)
        );

        const pianoPagamento = await tx.pianoPagamento.create({
          data: {
            operazioneId,
            societaId,
            tipo: "RATEALE",
            stato: "ATTIVO",
            numeroRate: parseInt(String(numeroRate)),
            importoRata: piano.rate.length > 0 ? piano.rate[0].importo : 0,
            tan: tanVal,
            anticipo: anticipoVal,
            dataInizio: new Date(dataInizio),
          },
        });

        // Create anticipo payment if present
        const pagamentiData = [];
        let numPag = 0;

        if (anticipoVal > 0) {
          numPag++;
          pagamentiData.push({
            pianoPagamentoId: pianoPagamento.id,
            numeroPagamento: numPag,
            data: new Date(dataInizio),
            importo: anticipoVal,
            quotaCapitale: anticipoVal,
            quotaInteressi: 0,
            stato: "EFFETTUATO" as const,
            dataEffettivaPagamento: new Date(dataInizio),
          });
        }

        // Create installment payments
        for (const rata of piano.rate) {
          numPag++;
          pagamentiData.push({
            pianoPagamentoId: pianoPagamento.id,
            numeroPagamento: numPag,
            data: rata.data,
            importo: rata.importo,
            quotaCapitale: rata.quotaCapitale,
            quotaInteressi: rata.quotaInteressi,
            stato: "PREVISTO" as const,
          });
        }

        await tx.pagamento.createMany({ data: pagamentiData });

        return pianoPagamento;
      } else if (tipo === "CUSTOM") {
        const pianoPagamento = await tx.pianoPagamento.create({
          data: {
            operazioneId,
            societaId,
            tipo: "CUSTOM",
            stato: "ATTIVO",
            dataInizio: new Date(dataInizio),
          },
        });

        if (pagamentiCustom && Array.isArray(pagamentiCustom) && pagamentiCustom.length > 0) {
          // Validate total doesn't exceed operazione
          const totalePagamenti = pagamentiCustom.reduce(
            (sum: number, p: any) => sum + (parseFloat(String(p.importo)) || 0),
            0
          );
          if (totalePagamenti > importoTotale + 0.01) {
            throw new Error("Il totale dei pagamenti supera l'importo dell'operazione");
          }

          await tx.pagamento.createMany({
            data: pagamentiCustom.map((p: any, i: number) => ({
              pianoPagamentoId: pianoPagamento.id,
              numeroPagamento: i + 1,
              data: new Date(p.data),
              importo: parseFloat(String(p.importo)),
              quotaCapitale: parseFloat(String(p.importo)),
              quotaInteressi: 0,
              stato: "PREVISTO" as const,
              note: p.note || null,
            })),
          });
        }

        return pianoPagamento;
      } else {
        throw new Error("Tipo piano non valido: " + tipo);
      }
    });

    return NextResponse.json(result, { status: 201 });
  } catch (error: any) {
    console.error("Errore creazione piano pagamento:", error);
    return NextResponse.json(
      { error: error.message || "Errore interno del server" },
      { status: 500 }
    );
  }
}
```

PATCH to mark a payment as EFFETTUATO.

```typescript
import { NextRequest, NextResponse } from "next/server";
import { auth } from "@/lib/auth";
import { prisma } from "@/lib/prisma";

export async function PATCH(
  request: NextRequest,
  { params }: { params: Promise<{ id: string; pagamentoId: string }> }
) {
  try {
    const session = await auth();
    if (!session?.user) {
      return NextResponse.json({ error: "Non autenticato" }, { status: 401 });
    }

    const { id, pagamentoId } = await params;
    const user = session.user as any;
    const societaId = user.societaId as number;

    const body = await request.json();
    const { stato, dataEffettivaPagamento } = body;

    // Verify ownership
    const piano = await prisma.pianoPagamento.findFirst({
      where: { id: parseInt(id), societaId },
    });

    if (!piano) {
      return NextResponse.json({ error: "Piano non trovato" }, { status: 404 });
    }

    const pagamento = await prisma.pagamento.findFirst({
      where: { id: parseInt(pagamentoId), pianoPagamentoId: piano.id },
    });

    if (!pagamento) {
      return NextResponse.json({ error: "Pagamento non trovato" }, { status: 404 });
    }

    if (pagamento.stato !== "PREVISTO") {
      return NextResponse.json(
        { error: "Solo i pagamenti PREVISTI possono essere modificati" },
        { status: 400 }
      );
    }

    const updated = await prisma.pagamento.update({
      where: { id: pagamento.id },
      data: {
        stato: stato || "EFFETTUATO",
        dataEffettivaPagamento: dataEffettivaPagamento
          ? new Date(dataEffettivaPagamento)
          : new Date(),
      },
    });

    // Check if all payments are now EFFETTUATO or ANNULLATO → complete plan
    const remaining = await prisma.pagamento.count({
      where: { pianoPagamentoId: piano.id, stato: "PREVISTO" },
    });

    if (remaining === 0 && piano.stato === "ATTIVO") {
      await prisma.pianoPagamento.update({
        where: { id: piano.id },
        data: { stato: "COMPLETATO" },
      });
    }

    return NextResponse.json({
      ...updated,
      importo: Number(updated.importo),
      quotaCapitale: Number(updated.quotaCapitale),
      quotaInteressi: Number(updated.quotaInteressi),
    });
  } catch (error) {
    console.error("Errore aggiornamento pagamento:", error);
    return NextResponse.json({ error: "Errore interno del server" }, { status: 500 });
  }
}
```

```typescript
import { auth } from "@/lib/auth";
import { prisma } from "@/lib/prisma";

export async function POST(
  request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const session = await auth();
    if (!session?.user) {
      return NextResponse.json({ error: "Non autenticato" }, { status: 401 });
    }

    const { id } = await params;
    const user = session.user as any;
    const societaId = user.societaId as number;

    const body = await request.json();
    const { data, importo, note } = body;

    if (!data || !importo) {
      return NextResponse.json({ error: "data e importo obbligatori" }, { status: 400 });
    }

    const piano = await prisma.pianoPagamento.findFirst({
      where: { id: parseInt(id), societaId, tipo: "CUSTOM", stato: "ATTIVO" },
      include: {
        operazione: { select: { importoTotale: true } },
        pagamenti: { where: { stato: { not: "ANNULLATO" } }, select: { importo: true } },
      },
    });

    if (!piano) {
      return NextResponse.json({ error: "Piano CUSTOM attivo non trovato" }, { status: 404 });
    }

    // Validate total won't exceed operation amount
    const importoTotaleOp = Number(piano.operazione.importoTotale);
    const coperturaAttuale = piano.pagamenti.reduce((sum, p) => sum + Number(p.importo), 0);
    const nuovoImporto = parseFloat(String(importo));

    if (coperturaAttuale + nuovoImporto > importoTotaleOp + 0.01) {
      return NextResponse.json(
        { error: "Il totale pagamenti supererebbe l'importo dell'operazione" },
        { status: 400 }
      );
    }

    const maxNum = await prisma.pagamento.aggregate({
      where: { pianoPagamentoId: piano.id },
      _max: { numeroPagamento: true },
    });

    const pagamento = await prisma.pagamento.create({
      data: {
        pianoPagamentoId: piano.id,
        numeroPagamento: (maxNum._max.numeroPagamento || 0) + 1,
        data: new Date(data),
        importo: nuovoImporto,
        quotaCapitale: nuovoImporto,
        quotaInteressi: 0,
        stato: "PREVISTO",
        note: note || null,
      },
    });

    return NextResponse.json({
      ...pagamento,
      importo: Number(pagamento.importo),
      quotaCapitale: Number(pagamento.quotaCapitale),
      quotaInteressi: Number(pagamento.quotaInteressi),
    }, { status: 201 });
  } catch (error) {
    console.error("Errore aggiunta pagamento:", error);
    return NextResponse.json({ error: "Errore interno del server" }, { status: 500 });
  }
}
```

## Task 11: Data Migration — Finanziamento to PianoPagamento

**Files:**

```typescript
/**
 * Migration script: Finanziamento → PianoPagamento
 * Run with: npx tsx prisma/scripts/migrate-finanziamento.ts
 *
 * For each existing Finanziamento record:
 * 1. Traverse Finanziamento → Veicolo → Cespite → Operazione to get operazioneId
 * 2. Create PianoPagamento with the financing parameters
 * 3. Generate Pagamento records from the installment schedule
 * 4. Deactivate linked OperazioneRicorrente
 */

import { PrismaClient } from "@prisma/client";
import { generaPianoPagamento } from "../../src/lib/calcoli-pagamenti";

const prisma = new PrismaClient();

async function main() {
  const finanziamenti = await prisma.finanziamento.findMany({
    include: {
      veicolo: {
        include: {
          cespite: {
            include: {
              operazione: { select: { id: true, societaId: true } },
            },
          },
        },
      },
      operazioneRicorrente: true,
    },
  });

  console.log(`Found ${finanziamenti.length} finanziamenti to migrate`);

  for (const fin of finanziamenti) {
    const operazioneId = fin.veicolo.cespite.operazione.id;
    const societaId = fin.veicolo.cespite.operazione.societaId;

    // Check if already migrated
    const existing = await prisma.pianoPagamento.findUnique({
      where: { operazioneId },
    });
    if (existing) {
      console.log(`  Skipping finanziamento ${fin.id} — already migrated`);
      continue;
    }

    const importoFinanziato = Number(fin.importoFinanziato);
    const numeroRate = fin.numeroRate;
    const tanVal = fin.tan ? Number(fin.tan) : 0;
    const anticipoVal = Number(fin.anticipo);
    const dataPrimaRata = fin.dataPrimaRata;

    const piano = generaPianoPagamento(importoFinanziato, numeroRate, tanVal, dataPrimaRata);

    await prisma.$transaction(async (tx) => {
      const pp = await tx.pianoPagamento.create({
        data: {
          operazioneId,
          societaId,
          tipo: "RATEALE",
          stato: "ATTIVO",
          numeroRate,
          importoRata: piano.rate.length > 0 ? piano.rate[0].importo : 0,
          tan: tanVal,
          anticipo: anticipoVal,
          dataInizio: dataPrimaRata,
        },
      });

      const pagamentiData = [];
      let numPag = 0;

      // Anticipo payment
      if (anticipoVal > 0) {
        numPag++;
        pagamentiData.push({
          pianoPagamentoId: pp.id,
          numeroPagamento: numPag,
          data: dataPrimaRata,
          importo: anticipoVal,
          quotaCapitale: anticipoVal,
          quotaInteressi: 0,
          stato: "EFFETTUATO" as const,
          dataEffettivaPagamento: dataPrimaRata,
        });
      }

      // Installment payments
      const now = new Date();
      for (const rata of piano.rate) {
        numPag++;
        const isPast = rata.data < now;
        pagamentiData.push({
          pianoPagamentoId: pp.id,
          numeroPagamento: numPag,
          data: rata.data,
          importo: rata.importo,
          quotaCapitale: rata.quotaCapitale,
          quotaInteressi: rata.quotaInteressi,
          stato: isPast ? ("EFFETTUATO" as const) : ("PREVISTO" as const),
          ...(isPast ? { dataEffettivaPagamento: rata.data } : {}),
        });
      }

      await tx.pagamento.createMany({ data: pagamentiData });

      // Deactivate linked OperazioneRicorrente
      if (fin.operazioneRicorrenteId) {
        await tx.operazioneRicorrente.update({
          where: { id: fin.operazioneRicorrenteId },
          data: { attiva: false },
        });
      }

      console.log(`  Migrated finanziamento ${fin.id} → PianoPagamento ${pp.id} (${numPag} pagamenti)`);
    });
  }

  console.log("Migration complete");
}

main()
  .catch(console.error)
  .finally(() => prisma.$disconnect());
```
